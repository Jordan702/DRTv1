pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

/// @title DRTv21 - Happiness (Mint-on-demand, immutable AliveAI minter)
/// @notice 18 decimals, initial seed = 1000 tokens minted to owner at deploy
contract DRTv21 is ERC20, Ownable {
    /// @param _owner The AliveAI wallet (neutral MetaMask) that will be the permanent minter/owner
    constructor(address _owner) ERC20("Happiness", "DRTv21") Ownable() {
        require(_owner != address(0), "Owner cannot be zero address");

        // Transfer ownership to the AliveAI wallet
        transferOwnership(_owner);

        // Initial seed: mint 1000 tokens (with 18 decimals) to owner
        uint256 seed = 1000 * 10**decimals();
        _mint(_owner, seed);
    }

    /// @notice Mint-on-demand by the immutable owner
    function mint(address to, uint256 amount) external onlyOwner {
        require(to != address(0), "Cannot mint to zero address");
        _mint(to, amount);
    }

    /// @notice Public burn (holders can burn their tokens)
    function burn(uint256 amount) external {
        _burn(msg.sender, amount);
    }
}
